sudo apt-get update && apt-get upgrade
apt install tmux
tmux -V
git clone https://github.com/tmux-plugins/tpm ~/.tmux/plugins/tpm
touch ~/.tmux.conf

vim ~/.tmux.conf:
# List of plugins
set -g @plugin 'tmux-plugins/tpm'
set -g @plugin 'tmux-plugins/tmux-sensible'

# Other examples:
# set -g @plugin 'github_username/plugin_name'
# set -g @plugin 'github_username/plugin_name#branch'
# set -g @plugin 'git@github.com:user/plugin'
# set -g @plugin 'git@bitbucket.com:user/plugin'

# Initialize TMUX plugin manager (keep this line at the very bottom of tmux.conf)
run '~/.tmux/plugins/tpm/tpm'

# type this in terminal if tmux is already running
tmux source ~/.tmux.conf

cheatsheet:

prefix key:
ctrl + b 

prefix -> shift + 5 or % //open window horizontally
-> " //open window vertically
-> hold prefix and arrow key to adjust the size
-> arrow keys to switch
-> { or } to switch cursor to next or previous screen
-> c or z//current window full screen
-> num(3) switch to 3rd terminal
-> n or p (next or previous)
-> & (kill window)
